</Short>



<Type>
 string|number|"content"


</Type>


~~~js
 
var form = new dhx.Form("form_container", {
	css: "dhx_widget--bordered",
	width: "500px", /*!*/ // the width for rows
    rows: [
        {
            width: "500px", /*!*/ // the width for cols
            css: "dhx_layout-cell--bordered",
            cols: [{
                type: "checkbox",
                label: "start",
                name: "align",
                labelInline: true,
                value: "start",
                checked: true
            },
            {
                type: "checkbox",
                label: "center",
                name: "align",
                labelInline: true,
                value: "center"
            }]
        }
    ]
});


~~~
