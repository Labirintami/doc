<Short>
 
adds style classes to all containers of Sidebar controls with nested items 





</Short>



<Type>
 string


</Type>


~~~js
 
<style>
    .my_first_class {
        /*some styles*/
    }
 
    .my_second_class {
        /*some styles*/
    }
</style>

var sidebar = new dhx.Sidebar("sidebar", {
    menuCss:"my_first_class my_second_class"
});


~~~
