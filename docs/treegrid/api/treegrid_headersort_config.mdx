<Short>
 
defines whether sorting on clicking headers of columns is enabled



</Short>


<strong>Default value: </strong> 
true



<Type>
 boolean


</Type>


~~~js
 
var grid = new dhx.Grid("grid_container", {
	columns: [// columns config],
	headerSort:false,  /*!*/
	data: dataset
});


~~~
