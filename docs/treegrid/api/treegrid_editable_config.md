editable
=============

@short: 
enables editing in TreeGrid columns


@default:
false


@type: boolean

@example: 
var treegrid = new dhx.TreeGrid("treegrid_container", {
	columns: [// columns config],
	editable:true, /*!*/
	data: dataset
});


@template:	api_config
