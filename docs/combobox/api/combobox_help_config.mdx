import { Short, Type, Descr } from '@site/src/components';

          


<Short>
 
adds an icon with a question mark next to the Combo input



</Short>



<Type>
 string


</Type>


~~~js
 
var combobox = new dhx.Combobox("combo_container",{
	help:"Some text"
});


~~~
	api_config

<Descr>
 
The property is useful while using Combobox in form.



</Descr>


 instead of it, you can use the combo/api/combobox_helpmessage_config.md property:

~~~js
var combobox = new dhx.Combobox("combo_container",{
	helpMessage:"Some text"
});
~~~
