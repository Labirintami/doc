</Short>



<Type>
 boolean


</Type>


~~~js
 
var source = new dhx.DataView("dataview-source", {dragMode: "source", dragCopy: true});
var target = new dhx.DataView("dataview-target", {dragMode: "target", dragCopy: true});


~~~
