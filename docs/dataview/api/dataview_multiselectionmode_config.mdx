<Short>
 
sets the mode of multiselection



</Short>


<strong>Default value: </strong> 
"ctrlClick"
"click","ctrlClick"


<Type>
 string


</Type>


~~~js
 
var dataview = new dhx.DataView("dataview", {
	multiselection:true,
	multiselectionMode:"click"   
});


~~~
