<Short>
 defines the format of the value to be applied when working with TimePicker events



</Short>



<Type>
 string


</Type>


 "string", "timeObject"

~~~js
 
var timepicker = new dhx.Timepicker("timepicker", {
	value: "06:00",
	timeFormat: 24,
	valueFormat: "string"
});

// or
var timepicker = new dhx.Timepicker("timepicker", {
	value: {hour: 6, minute: 0, AM: true},
	timeFormat: 12,
	valueFormat: "timeObject"
});


~~~
