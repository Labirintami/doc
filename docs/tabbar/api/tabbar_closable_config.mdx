<Short>
 adds close buttons for tabs






</Short>



<Type>
 boolean|string[]


</Type>


~~~js
 
// adds close buttons for all tabs
var tabbar = new dhx.Tabbar("tabbar", {
	closable: true
});

// adds close buttons for separate tabs
var tabbar = new dhx.Tabbar("tabbar", {
	closable: ["paris", "london"]
});

~~~
