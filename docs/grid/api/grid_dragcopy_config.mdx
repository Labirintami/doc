<Short>
 
defines that a row is copied to a target during drag-n-drop





</Short>



<Type>
 boolean


</Type>


~~~js
 
var source = new dhx.Grid("grid-source", {
    columns: [// columns config],
    dragMode: "source", 
    dragCopy: true
});
var target = new dhx.Grid("grid-target", {
    columns: [// columns config],
    dragMode: "target", 
    dragCopy: true
});
~~~
