<Short>
 defines the type of borders between cells inside a layout



</Short>



<Type>
 string


</Type>


 "line", "space", "wide", "none"

~~~js
 
const layout = new dhx.Layout("layout", {
    type: "space", /*!*/
    cols: [
        {
            html: "1"
        },
        {
            html: "2"
        },
        {
            html: "3"
        }
    ]
});


~~~
