<Short>
 
adds style classes to all containers of Toolbar controls with nested items 


</Short>



<Type>
 string


</Type>


~~~js
 
<style>
    .my_first_class {
        /*some styles*/
    }
 
    .my_second_class {
        /*some styles*/
    }
</style>
 
var toolbar = new dhx.Toolbar({
    menuCss:"my_first_class my_second_class"
});


~~~
