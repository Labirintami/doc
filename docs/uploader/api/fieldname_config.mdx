<Short>
  
	sets the name of the file field in the form data

</Short>


<strong>Default value: </strong> "file"

<Type>
 string

</Type>


~~~js

var vault = new dhx.Vault("vault_container", { 
    uploader:{
    	// obligatory, the path for upload
    	target:"/upload"
    	fieldName:"document"
    }
});
~~~
